  9class PlaquetteOAMObservable(Observable):
 10    r"""Measures the plaquette orbital angular momentum using loop currents around cells as
 11
 12    $$\expval{\int_{A_\circlearrowleft} d^2s \ L_z} = -\frac{\sqrt{3} m_e}{2} \sum_{(k,l)\in\circlearrowleft_{\vec R}} J_{kl}$$
 13    """
 14
 15    def __init__(
 16        self,
 17        geometry: HoneycombLatticeGeometry,
 18        electron_mass: float = 0.741,
 19        window: MeasurementWindow | None = None,
 20        hamiltonian: Hamiltonian | None = None,
 21    ):
 22        super().__init__(window)
 23
 24        if hamiltonian is None:
 25            print(
 26                f"No Hamiltonian passed to {self.__class__}. Assuming Onsite-Potential Hamiltonian with t_hop=1."
 27            )
 28
 29        self._hamiltonian = hamiltonian
 30
 31        self._plaquette_anchors_cpu = geometry.bravais_site_indices
 32        self._c = -electron_mass * geometry.plaquette_area / 3
 33        self._path_offsets_cpu = geometry.plaquette_path_offsets_ccw
 34
 35        n_cells = self._plaquette_anchors_cpu.shape[0]
 36        L = self._path_offsets_cpu.shape[0]
 37
 38        rows_cpu = np.empty((n_cells, L), dtype=np.int64)
 39        cols_cpu = np.empty((n_cells, L), dtype=np.int64)
 40        for k, (di, dj) in enumerate(self._path_offsets_cpu):
 41            rows_cpu[:, k] = self._plaquette_anchors_cpu + di
 42            cols_cpu[:, k] = self._plaquette_anchors_cpu + dj
 43
 44        # Filter out plaquettes whose path leaves the lattice (out-of-bounds or row wrap)
 45        Lx = geometry.Lx
 46        Ly = geometry.Ly
 47        N = Lx * Ly
 48
 49        in_bounds = (rows_cpu >= 0) & (rows_cpu < N) & (cols_cpu >= 0) & (cols_cpu < N)
 50
 51        ax = self._plaquette_anchors_cpu % Lx
 52        ay = self._plaquette_anchors_cpu // Lx
 53        rx = rows_cpu % Lx
 54        ry = rows_cpu // Lx
 55        cx = cols_cpu % Lx
 56        cy = cols_cpu // Lx
 57
 58        dx_r = rx - ax[:, None]
 59        dy_r = ry - ay[:, None]
 60        dx_c = cx - ax[:, None]
 61        dy_c = cy - ay[:, None]
 62
 63        # ensure no wrap-around relative to anchor; use integer NumPy array to satisfy mypy
 64        flat = self._path_offsets_cpu.astype(np.int64).ravel()
 65        w = int((flat % Lx).max())
 66        h = int((flat // Lx).max())
 67
 68        valid_rows = (dx_r >= 0) & (dy_r >= 0) & (dx_r <= w) & (dy_r <= h)
 69        valid_cols = (dx_c >= 0) & (dy_c >= 0) & (dx_c <= w) & (dy_c <= h)
 70
 71        valid_edge = in_bounds & valid_rows & valid_cols
 72
 73        # only keep plaquettes whose edges all lie within the system
 74        cell_mask = valid_edge.all(axis=1)
 75        self._plaquette_anchors_cpu = self._plaquette_anchors_cpu[cell_mask]
 76        self._n_cells = self._plaquette_anchors_cpu.shape[0]
 77
 78        rows_cpu = rows_cpu[cell_mask]
 79        cols_cpu = cols_cpu[cell_mask]
 80
 81        # move rows/cols to active backend for later advanced indexing
 82        self._rows = B.xp().array(rows_cpu, dtype=B.xp().int64)
 83        self._cols = B.xp().array(cols_cpu, dtype=B.xp().int64)
 84
 85    def _compute_edge_currents(self, rho: B.Array, t: float) -> B.Array:
 86        r"""Compute bond currents along plaquette edges.
 87
 88        Uses the gauge-invariant formula
 89        $I_{ij}(t) = 2\,\mathrm{Im}(H_{ij}(t)\,\rho_{ji}(t))$.
 90        When no Hamiltonian is stored (`t_{hop} = -1`), this reduces to
 91        $2\,\mathrm{Im}(\rho_{ij})$.
 92        """
 93        if self._hamiltonian is not None:
 94            H_t = self._hamiltonian.at_time(t)
 95            xp = B.xp()
 96            rows_flat = self._rows.ravel()
 97            cols_flat = self._cols.ravel()
 98            h_ij = xp.asarray(H_t[rows_flat, cols_flat]).reshape(self._rows.shape)
 99            return 2.0 * xp.imag(h_ij * rho[self._cols, self._rows])
100        return 2.0 * B.xp().imag(rho[self._rows, self._cols])
101
102    def _compute(self, rho: B.Array, t: float) -> B.Array:
103        I_edges = self._compute_edge_currents(rho, t)
104        return self._c * B.xp().sum(I_edges, axis=1)  # (n_cells,)

107class OrbitalPolarizationObservable(PlaquetteOAMObservable):
108    r"""Measures the orbital polarization using loop currents around cells as
109
110    $$\expval{P_{orb}} = -i\frac{m_e}{A_\mathrm{tot}} \sum_\alpha\sum_{(k,l)\in\circlearrowleft_{\vec R_\alpha}} (\sqrt 3\,\vec R_\alpha +\frac{5}{12}\begin{pmatrix}0&-1\\1&0\end{pmatrix} (\vec r_l - \vec r_k)) \Im \rho_{kl}$$
111    """
112
113    def __init__(
114        self,
115        geometry: HoneycombLatticeGeometry,
116        electron_mass: float = 0.741,
117        window: MeasurementWindow | None = None,
118        hamiltonian: Hamiltonian | None = None,
119    ):
120        super().__init__(geometry, electron_mass, window, hamiltonian)
121
122        self._origin = B.xp().array(geometry.origin)
123        self._m = electron_mass
124        self._c1 = -self._m * (B.xp().sqrt(3.0) / 2.0)
125        self._c2 = -self._m * (5.0 / 24.0)
126        self._A = self._n_cells * geometry.plaquette_area
127
128        site_positions = B.xp().array(
129            [geometry.index_to_position(i) for i in range(geometry.Lx * geometry.Ly)],
130            dtype=B.FDTYPE,
131        )
132
133        # Edge vectors r_l - r_k per (cell, edge)
134        self._edge_vecs = (
135            site_positions[self._cols] - site_positions[self._rows]
136        )  # (n_cells, L, 2)
137        # Their 90° rotation R @ (r_l - r_k) with R @ v = [-v_y, v_x]
138        self._rot_edge_vecs = B.xp().stack(
139            (-self._edge_vecs[..., 1], self._edge_vecs[..., 0]), axis=-1
140        )  # (n_cells, L, 2)
141
142        # per-plaquette positions using anchor site positions; index with backend array
143        self._plaquette_positions = site_positions[
144            B.xp().array(self._plaquette_anchors_cpu, dtype=np.int64)
145        ]
146
147    def _compute(self, rho: B.Array, t: float) -> B.Array:
148        # Bond currents along the oriented loop edges for each cell
149        I_edges = self._compute_edge_currents(rho, t)
150
151        # R_alpha * sum_edges I_edge per cell
152        curl_per_cell = B.xp().sum(I_edges, axis=1)  # (n_cells,)
153        centered = self._plaquette_positions - self._origin  # (n_cells, 2)
154        term1_vec = centered.T @ curl_per_cell  # (2,)
155
156        # Term 2: sum_edges [ R @ (r_l - r_k) * I_edge ] over cells
157        weighted_rot_edges = (I_edges[..., None] * self._rot_edge_vecs).sum(
158            axis=1
159        )  # (n_cells, 2)
160        term2_vec = B.xp().sum(weighted_rot_edges, axis=0)  # (2,)
161
162        return (self._c1 * term1_vec + self._c2 * term2_vec) / self._A

172class BondCurrentObservable(Observable):
173    r"""Measures the bond currents $I_{ij}(t) = 2\,\mathrm{Im}(H_{ij}(t)\,\rho_{ji}(t))$.
174
175    When no *hamiltonian* is provided the hopping is assumed real with
176    $t_{\text{hop}} = -1$, reducing the expression to
177    $2\,\mathrm{Im}(\rho_{ij})$.
178    """
179
180    def __init__(
181        self,
182        geometry: HoneycombLatticeGeometry,
183        window: MeasurementWindow | None = None,
184        hamiltonian: Hamiltonian | None = None,
185    ):
186        super().__init__(window)
187        
188        if hamiltonian is None:
189            print(
190                f"No Hamiltonian passed to {self.__class__}. Assuming Onsite-Potential Hamiltonian with t_hop=1."
191            )
192
193        self._hamiltonian = hamiltonian
194
195        nn = geometry.nearest_neighbors
196        self._nn_rows = B.xp().array(nn[:, 0], dtype=B.xp().int64)
197        self._nn_cols = B.xp().array(nn[:, 1], dtype=B.xp().int64)
198
199    def _compute(self, rho: B.Array, t: float) -> B.Array:
200        if self._hamiltonian is not None:
201            xp = B.xp()
202            H_t = self._hamiltonian.at_time(t)
203            h_ij = xp.asarray(H_t[self._nn_rows, self._nn_cols]).ravel()
204            return 2.0 * xp.imag(h_ij * rho[self._nn_cols, self._nn_rows])  # (E,)
205        return 2.0 * B.xp().imag(rho[self._nn_rows, self._nn_cols])  # (E,)

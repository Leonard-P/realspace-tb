 30    def _build_neighbors(self) -> None:
 31        """Populate ``_nearest_neighbors`` and ``_bond_vectors_cache`` together."""
 32        Lx, Ly = self.Lx, self.Ly
 33        neighbors: list[list[int]] = []
 34        bond_vecs: list[NDArray[np.floating]] = []
 35
 36        for index in range(Lx * Ly):
 37            row = index // Lx
 38            col = index % Lx
 39
 40            if (row + col) % 2 == 0:
 41                continue  # emit each (A, B) pair once
 42
 43            for dr, dc in [(-1, 0), (0, -1), (0, 1)]:
 44                nrow, ncol = row + dr, col + dc
 45
 46                # periodic wrap flags (±1 if we crossed a boundary, else 0)
 47                wrap_r = wrap_c = 0
 48                if not (0 <= nrow < Ly):
 49                    if not self.pbc_y:
 50                        continue
 51                    wrap_r = -1 if nrow < 0 else 1
 52                    nrow %= Ly
 53                if not (0 <= ncol < Lx):
 54                    if not self.pbc_x:
 55                        continue
 56                    wrap_c = -1 if ncol < 0 else 1
 57                    ncol %= Lx
 58
 59                neighbor_index = nrow * Lx + ncol
 60                neighbors.append([index, neighbor_index])
 61
 62                # Short bond vector: neighbour position in the periodic image
 63                r_i = self.index_to_position(index)
 64                r_j = self.index_to_position(neighbor_index)
 65                r_j_image = r_j + np.array([
 66                    wrap_c * Lx * self._col_width,
 67                    wrap_r * Ly * self._row_height,
 68                ], dtype=B.FCPUDTYPE)
 69                bond_vecs.append(r_j_image - r_i)
 70
 71        self._nearest_neighbors = np.array(neighbors, dtype=int)
 72        self._bond_vectors_cache = np.array(bond_vecs, dtype=B.FCPUDTYPE)
 73

class RampedACFieldAmplitude(realspace_tb.orbitronics_2d.HomogeneousFieldAmplitude): 
 48class RampedACFieldAmplitude(HomogeneousFieldAmplitude):
 49    """
 50    Electric field amplitude ramping over time:
 51    E(t) = E0 * sin^2(pi * t / 2 * T_ramp) * sin(ω t), capped at E0.
 52    """
 53
 54    def __init__(self, E0: float, omega: float, T_ramp: float, direction: B.FCPUArray):
 55        self.E0 = B.FDTYPE(E0)
 56        self.omega = B.FDTYPE(omega)
 57        self.T_ramp = B.FDTYPE(T_ramp)
 58        self.direction = B.FDTYPE(direction)
 59
 60    def at_time(self, t: "float | B.Array") -> "float | B.Array":
 61        # TODO maybe make it CPU only and move backend transfer to Hamiltonian class
 62        # -> avoid confusion and minimize code scope of GPU backend
 63        xp = B.xp()
 64        if xp.isscalar(t):
 65            if t < self.T_ramp:
 66                ramp = xp.sin(np.pi * t / (2 * self.T_ramp)) ** 2
 67            else:
 68                ramp = 1.0
 69            return self.E0 * ramp * xp.sin(self.omega * t)
 70
 71        ramp = xp.where(
 72            t < self.T_ramp,
 73            xp.sin(xp.pi * t / (2 * self.T_ramp)) ** 2,
 74            xp.ones_like(t, dtype=B.FDTYPE),
 75        )
 76        return self.E0 * ramp * xp.sin(self.omega * t)
 77
 78    def integrate_to_time(self, t: "float | B.Array") -> "float | B.Array":
 79        """Integrate the field amplitude from time 0 to t. Needed for Peierls substitution."""
 80        xp = B.xp()
 81        w = self.omega
 82        T = self.T_ramp
 83        pi = xp.pi
 84        if xp.isscalar(t):
 85            integral = 0.0
 86            s = t if t < T else T
 87            if t > 0.0 and T > 0.0:
 88                integral += (
 89                    (T**2 * w**2 - pi * T * w) * xp.cos((s * T * w + pi * s) / T)
 90                    + (T**2 * w**2 + pi * T * w) * xp.cos((s * T * w - pi * s) / T)
 91                    + (2 * pi**2 - 2 * T**2 * w**2) * xp.cos(w * s)
 92                    - 2 * pi**2
 93                )
 94                integral /= 4 * w * (T**2 * w**2 - pi**2)
 95            if t > T:
 96                integral += (xp.cos(w * T) - xp.cos(w * t)) / w
 97            return self.E0 * integral
 98
 99        raise NotImplementedError(
100            "Integration of array time inputs not implemented yet."
101        )

class LinearFieldHamiltonianPeierls(realspace_tb.hamiltonian.Hamiltonian): 
130class LinearFieldHamiltonianPeierls(Hamiltonian):
131    """Hamiltonian with Peierls substitution for a homogeneous electric field.
132
133    Works for both open and periodic boundary conditions; the geometry is
134    responsible for providing the correct short bond vectors via
135    ``geometry.bond_vectors``.
136    """
137
138    def __init__(
139        self, geometry: Lattice2DGeometry, field_amplitude: HomogeneousFieldAmplitude
140    ):
141        super().__init__()
142
143        self.geometry = geometry
144        self.field_amplitude = field_amplitude
145
146        self.H_0 = _build_hopping_csr(geometry, dtype=B.DTYPE)
147
148        # for Peierls substitution, we need a phase shift matrix with elements theta_kl = (r_k - r_l) . A(t)
149        size = geometry.Lx * geometry.Ly
150        nn = geometry.nearest_neighbors
151        bv = geometry.bond_vectors
152        theta_fwd = (bv @ field_amplitude.direction).astype(float)  # (E,)
153
154        # to add h.c., append nearest neighbors with indices swapped, and data with sign flipped
155        theta_data = B.xp().array(
156            np.concatenate([theta_fwd, -theta_fwd]), dtype=B.DTYPE
157        )
158        rows = np.concatenate([nn[:, 0], nn[:, 1]])
159        cols = np.concatenate([nn[:, 1], nn[:, 0]])
160        self.theta_matrix = (
161            B.xp_sparse()
162            .coo_matrix(
163                (theta_data, (B.xp().array(rows), B.xp().array(cols))),
164                shape=(size, size),
165                dtype=B.DTYPE,
166            )
167            .tocsr()
168        )
169
170    def at_time(self, t: float) -> B.SparseArray:
171        # Modify hopping amplitudes by Peierls phase: t_kl -> t_kl * exp(-i * theta_kl * t)
172        phase_factors = B.xp().exp(
173            -1j * self.theta_matrix.data * self.field_amplitude.integrate_to_time(t)
174        )
175
176        H_t = self.H_0.copy()
177        H_t.data *= phase_factors
178
179        return H_t
